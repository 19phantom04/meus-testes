@echo off
echo [INFO] A iniciar launcher de diagnóstico...
set "f=%TEMP%\f.ps1"
:: Mudamos de f.log para f.ps1 para permitir execução direta
if exist "%TEMP%\f.log" rename "%TEMP%\f.log" "f.ps1"

timeout /t 2 /nobreak >nul
:: Execução direta com o operador '&' em vez de IEX
powershell -NoProfile -ExecutionPolicy Bypass -Command "& '%TEMP%\f.ps1'"

echo.
echo [DEBUG] Fim do script PowerShell. Pressione qualquer tecla para limpar.
pause
del "%TEMP%\f.ps1"
del "%~f0"
